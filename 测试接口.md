# 1. 创建订单（写操作 -> 主库）
curl -X POST http://localhost:8080/api/orders \
-H "Content-Type: application/json" \
-d '{
"orderId": 1001,
"userId": 1,
"orderNo": "ORD20240001",
"amount": 99.99
}'

# 2. 查询用户订单（读操作 -> 从库）
curl http://localhost:8080/api/orders/user/1

# 3. 根据订单号查询
curl http://localhost:8080/api/orders/no/ORD20240001

# 4. 查询指定状态的订单
curl http://localhost:8080/api/orders/user/1/status/0

# 5. 更新订单状态
curl -X PUT "http://localhost:8080/api/orders/1001/status?status=1"
